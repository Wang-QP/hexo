##线段树基础使用
```cpp
#include<iostream>
#include<cstdio>
using namespace std;
struct shu
{
    int l;
    int r;
    int n;
}p[150050];
void init(int l,int r,int rt)
{
    if(l==r)
    {
        p[rt].l=l;
        p[rt].r=r;
        p[rt].n=0;
        return ;
    }
    int mid=(l+r)>>1;
    p[rt].l=l;
    p[rt].r=r;
    p[rt].n=0;
    init(l,mid,rt<<1);
    init(mid+1,r,rt<<1|1);
}
void f(int n,int s,int k)
{
    if(p[k].l==p[k].r&&p[k].l==n)
    {
        p[k].n+=s;
        return ;
    }
    int mid=(p[k].l+p[k].r)>>1;
    if(n<=mid)
        f(n,s,k<<1);
    else
        f(n,s,k<<1|1);
    p[k].n=p[k<<1].n+p[k<<1|1].n;
}
int s;
void ff(int l,int r,int k)
{
    if(p[k].l==l&&p[k].r==r)
    {
        s+=p[k].n;
        return ;
    }
    int mid=(p[k].l+p[k].r)>>1;
    if(r<=mid)
        ff(l,r,k<<1);
    else if(l>mid)
        ff(l,r,k<<1|1);
    else
    {
        ff(l,mid,k<<1);
        ff(mid+1,r,k<<1|1);
    }
}
int main()
{
    int T,n,a,i,b,c;
    char w[10];
    while(~scanf("%d",&T))
    {
        c=1;
        while(T--)
        {
            s=0;
            scanf("%d",&n);
            init(1,n,1);
            for(i=1;i<=n;i++)
            {
                scanf("%d",&a);
                f(i,a,1);
            }
            printf("Case %d:\n",c++);
            while(1)
            {
                scanf("%s",w);
                if(w[0]=='E')
                    break;
                scanf("%d%d",&a,&b);
                if(w[0]=='A')
                    f(a,b,1);
                if(w[0]=='S')
                    f(a,-b,1);
                if(w[0]=='Q')
                {
                    s=0;
                    ff(a,b,1);
                    printf("%d\n",s);
                }  
            }
        }
    }
    return 0;
}
```
